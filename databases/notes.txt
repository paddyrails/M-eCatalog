docker run -d \
  --name cs-sales \
  -e POSTGRES_DB=cs-sales \
  -e POSTGRES_USER=cs_sales_user \
  -e POSTGRES_PASSWORD=cs_sales_user \
  -v $(pwd)/init.sql:/docker-entrypoint-initdb.d/init.sql \
  -p 5432:5432 \
  postgres:latest


  docker exec -it cs-sales psql -U cs_sales_user -c "SHOW DATABASES;"


  for i in {1..30}; do curl -s -o /dev/null -w "%{http_code}\n" "http://localhost:9000/api/catalog/api/sales/catalog/v1/products?country=CAN"; done

  kubectl rollout restart deployment/api-gateway -n cs-sales
